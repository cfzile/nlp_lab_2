Предобработка данных¶

In [ ]:
texts = [article['text'] for article in articles]
categories = [article['category'] for article in articles]

  • Заменяем на пробелы все специальных символов (то есть оставляем только буквы и цифры) r'\W'

  • Заменяем на пробелы все одиночные символы (кроме цифр) r'\s+[a-zA-Z]\s+'

  • Заменяем на пробелы одиночный символ из начале текста (кроме цифр)r'\^[a-zA-Z]\s+'

  • Заменяем все несколько пробелов подряд на один пробел r'\s+'

  • Переводим весь текст в нижний регистр

  • Разбиваем на слова

  • Лемматизируем с помощью WordNetLemmatizer()

  • Удалять стоп слова будем при векторизации

Так же подсчитываем кол-во вхождений слов для каждой категории. (для графика)

In [ ]:
stemmer = WordNetLemmatizer()

x = []
words = {'arts': {}, 'sports': {}, 'politics': {}, 'business': {}}
stopwords_set = set(stopwords.words('english'))

for sen in tqdm(range(0, len(texts))):
    text = re.sub(r'\W', ' ', str(texts[sen]))
    text = re.sub(r'\s+[a-zA-Z]\s+', ' ', text)
    text = re.sub(r'\^[a-zA-Z]\s+', ' ', text) 
    text = re.sub(r'\s+', ' ', text)
    text = text.lower()
    
    text = text.split()
    text = [stemmer.lemmatize(word) for word in text]

    for word in text:
        if word in stopwords_set:
            continue
        if word not in words[categories[sen]]:
            words[categories[sen]][word] = 0
        else:
            words[categories[sen]][word] += 1

    x.append(' '.join(text))

In [ ]:
classifiers = [KNeighborsClassifier(), SVC(), RidgeClassifier(),
               RandomForestClassifier(), DecisionTreeClassifier(),
               MultinomialNB(), MLPClassifier(), AdaBoostClassifier()]

vectorizers = [TfidfVectorizer(stop_words = stopwords.words('english')),
               CountVectorizer(stop_words = stopwords.words('english'))]

param_grid_vectorizer = {
  'vectorizer__max_features': [1000, 1500],
  'vectorizer__max_df': [0.7, 1]
}

param_grids = [
    [{'classifier__n_neighbors': [3, 5, 8], **param_grid_vectorizer}], # KNeighborsClassifier
    [{'classifier__kernel': ['sigmoid', 'linear'],
      'classifier__C': [0.1, 1, 10], **param_grid_vectorizer}], # SVC
    [{'classifier__alpha': [0.01, 0.1, 1], **param_grid_vectorizer}], # RidgeClassifier
    [{'classifier__n_estimators': [200, 500], 'classifier__max_depth' : [4, 6, 10],
      **param_grid_vectorizer}], # RandomForestClassifier
    [{'classifier__criterion': ['gini', 'entropy'], 'classifier__max_depth': [4, 6, 10],
      **param_grid_vectorizer}], # DecisionTreeClassifier
    [{'classifier__alpha': [1, 0.1, 0.01], **param_grid_vectorizer}], # MultinomialNB
    [{'classifier__alpha': [0.01, 0.0001, 0.1], 'classifier__activation': ['logistic', 'relu'],
      **param_grid_vectorizer}], # MLPClassifier
    [{'classifier__n_estimators': [25, 100, 50], **param_grid_vectorizer}], # AdaBoostClassifier
]

pipelines = []

for classifier, param_grid in zip(classifiers, param_grids):
    for vectorizer in vectorizers:
        pipeline = Pipeline(steps=[('vectorizer', vectorizer), ('classifier', classifier)])
        pipelines.append([pipeline, param_grid])

results = []

for pipeline, param_grid in tqdm(pipelines):
    start = time.time()
    search = GridSearchCV(pipeline, param_grid)
    search.fit(x_train, y_train)
    stop = time.time()
    result = {}
    result['search_time'] = stop - start
    result['best_params_'] = search.best_params_
    result['best_score_'] = search.best_score_
    result['best_estimator_'] = search.best_estimator_
    result['cv_results_'] = search.cv_results_
    results.append(result)

scores, fit_time, predict_time, names = [], [], [], []

for result in tqdm(results):
    estimator = result['best_estimator_']
    classifier = estimator.steps[1][1].__class__.__name__
    vectorizer = estimator.steps[0][1].__class__.__name__
    names.append(classifier + ' ' + vectorizer)

    report = classification_report(estimator.predict(x_test), y_test, output_dict=True)
    scores.append([np.mean([report['macro avg']['f1-score'], report['weighted avg']['f1-score']]),
                   np.mean([report['macro avg']['recall'],  report['weighted avg']['recall']]),
                   np.mean([report['macro avg']['precision'],  report['weighted avg']['precision']]),
                   report['accuracy']])  
    
    start = time.time()
    for i in range(5):
        estimator.predict(x_test)
    stop = time.time()
    predict_time.append((stop - start)/5.)
    
    start = time.time()
    for i in range(5):
        estimator.fit(x_train, y_train)
    stop = time.time()
    fit_time.append((stop - start)/5.)
